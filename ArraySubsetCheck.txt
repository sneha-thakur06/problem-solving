import java.util.*;
public class ArraySubset {
    public static boolean isPresent(int[] arr, int[] brr){
        Map<Integer,Boolean> hmp=new HashMap<>();
        for (int k : arr) {
            hmp.put(k, true);
        }
        for (int i : brr) {
            if (!hmp.containsKey(i)) {
               return false;
            }
        }
        return true;
    }
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        System.out.println("Enter the size of the array: ");
        int n=sc.nextInt();


        System.out.println("Enter the size of another array: ");
        int m=sc.nextInt();

        int[] arr=new int[n];
        System.out.println("Enter the elements for the array: ");
        for(int i=0;i<arr.length;i++){
            arr[i]=sc.nextInt();
        }

        int[] brr=new int[m];
        System.out.println("Enter the element for the another array: ");
        for(int j=0;j<brr.length;j++){
            brr[j]=sc.nextInt();
        }
      System.out.println(isPresent(arr,brr));



}
}
